USE DB_BIBLIOTECA

SELECT * FROM TBL_AUTORES
LEFT JOIN TBL_LIVRO
ON TBL_LIVRO.ID_AUTOR = TBL_AUTORES.ID_AUTOR
WHERE TBL_LIVRO.ID_AUTOR IS NULL

SELECT * FROM TBL_AUTORES
RIGHT JOIN TBL_LIVRO
ON TBL_LIVRO.ID_AUTOR = TBL_AUTORES.ID_AUTOR
WHERE TBL_LIVRO.ID_AUTOR IS NULL

SELECT * FROM TBL_AUTORES
FULL JOIN TBL_LIVRO
ON TBL_LIVRO.ID_AUTOR = TBL_AUTORES.ID_AUTOR


SELECT * FROM TBL_AUTORES
FULL JOIN TBL_LIVRO
ON TBL_LIVRO.ID_AUTOR = TBL_AUTORES.ID_AUTOR
WHERE TBL_LIVRO.ID_AUTOR IS NULL 
OR TBL_AUTORES.ID_AUTOR IS NULL
